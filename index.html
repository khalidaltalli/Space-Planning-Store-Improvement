/** * DELIVERED D.N. REPORT * Organized by section with merged D.N. cells for shared groups */ function generateDeliveredDNReport() { var ss = SpreadsheetApp.openById('1U_UULtDDI9gbBSbORba-0I9z5068vXGC2_ntpoHcGRA'); var checklist = ss.getSheetByName('Receiving Checklist'); if (!checklist) { SpreadsheetApp.getUi().alert('Receiving Checklist not found!'); return; } var data = checklist.getDataRange().getValues(); var headers = data[0].map(function(h){ return String(h).trim(); }); var branchCol = headers.indexOf('Branch Code'); var assetCol = headers.indexOf('Asset'); var sectionCol = headers.indexOf('Section'); var statusCol = headers.indexOf('Asset Status'); var dnNSCol = headers.indexOf('D.N. Not Signed'); if ([branchCol,assetCol,sectionCol,statusCol,dnNSCol].indexOf(-1) > -1) { SpreadsheetApp.getUi().alert('Required columns not found!'); return; } /* ── Section order ── */ var SECTION_ORDER = ['Construction','Shelves','Fridges','Office','Store Sections','Consumable Assets','Store Tools']; /* ── Shared D.N. groups ── */ /* Key: "Branch|GroupName" → shared DN link */ /* Groups: Shelves, Fridges, CCTV+Access Points, Office Chair+Office Desk */ var SHARED_GROUPS = { 'Shelves': ['Freebolt','Heavyrack','Heavy Rack'], 'Fridges': ['1 Door Chiller','2 Door Chiller','3 Door Chiller','1 Door Freezer','2 Door Freezer','3 Door Freezer'], 'CCTV+Access Points': ['CCTV','Access Points'], 'Office Furniture': ['Office Chair','Office Desk'] }; /* Build asset → group map */ var assetToGroup = {}; Object.keys(SHARED_GROUPS).forEach(function(grp) { SHARED_GROUPS[grp].forEach(function(a) { assetToGroup[a] = grp; }); }); /* ── Filter Delivered rows ── */ var delivered = []; for (var i = 1; i < data.length; i++) { if (String(data[i][statusCol]).trim() === 'Delivered') { delivered.push({ branch: String(data[i][branchCol]).trim(), asset: String(data[i][assetCol]).trim(), section: String(data[i][sectionCol]).trim(), dn: String(data[i][dnNSCol]).trim() }); } } if (delivered.length === 0) { SpreadsheetApp.getUi().alert('No delivered assets found!'); return; } /* ── Sort by section order then by asset ── */ delivered.sort(function(a, b) { var si = SECTION_ORDER.indexOf(a.section); var sj = SECTION_ORDER.indexOf(b.section); if (si === -1) si = 99; if (sj === -1) sj = 99; if (si !== sj) return si - sj; /* Within section, group shared assets together */ var gi = assetToGroup[a.asset] || a.asset; var gj = assetToGroup[b.asset] || b.asset; if (gi !== gj) return gi.localeCompare(gj); return a.branch.localeCompare(b.branch); }); /* ── Create or clear report tab ── */ var reportSheet = ss.getSheetByName('Delivered D.N. Report'); if (!reportSheet) reportSheet = ss.insertSheet('Delivered D.N. Report'); else reportSheet.clearContents(); /* Clear all merges */ reportSheet.getRange(1, 1, reportSheet.getMaxRows(), reportSheet.getMaxColumns()).breakApart(); /* ── Write headers ── */ var hdrs = ['Section','Branch Code','Asset','D.N. Not Signed','Generated On']; reportSheet.appendRow(hdrs); reportSheet.getRange(1,1,1,hdrs.length) .setFontWeight('bold').setBackground('#1a3a5c').setFontColor('#ffffff'); /* ── Write rows ── */ var timestamp = new Date().toLocaleString(); var rowNum = 2; var sectionColors = { 'Construction': '#e8f4fd', 'Shelves': '#e8fdf0', 'Fridges': '#fdf0e8', 'Office': '#f5e8fd', 'Store Sections': '#fde8e8', 'Consumable Assets':'#fdfde8', 'Store Tools': '#e8fdfd' }; /* Group by section then by shared group */ var sections = {}; delivered.forEach(function(row) { var sec = row.section || 'Other'; if (!sections[sec]) sections[sec] = {}; var grp = assetToGroup[row.asset] || row.asset; /* individual if not shared */ if (!sections[sec][grp]) sections[sec][grp] = []; sections[sec][grp].push(row); }); SECTION_ORDER.forEach(function(sec) { if (!sections[sec]) return; var secColor = sectionColors[sec] || '#f8f9fa'; var groups = sections[sec]; /* Section header row */ reportSheet.getRange(rowNum, 1, 1, hdrs.length) .merge().setValue(sec) .setFontWeight('bold').setBackground('#34495e').setFontColor('#ffffff') .setHorizontalAlignment('center'); rowNum++; Object.keys(groups).forEach(function(grp) { var rows = groups[grp]; var isShared = !!SHARED_GROUPS[grp]; /* true if group has shared DN */ var startRow = rowNum; /* Determine shared DN link for this branch group if shared */ /* Group further by branch for shared groups */ if (isShared) { /* Sub-group by branch */ var byBranch = {}; rows.forEach(function(r) { if (!byBranch[r.branch]) byBranch[r.branch] = []; byBranch[r.branch].push(r); }); Object.keys(byBranch).sort().forEach(function(branch) { var bRows = byBranch[branch]; var sharedDN = ''; bRows.forEach(function(r) { if (r.dn && r.dn.startsWith('http')) sharedDN = r.dn; }); var bStart = rowNum; bRows.forEach(function(r) { reportSheet.getRange(rowNum, 1).setValue(r.section); reportSheet.getRange(rowNum, 2).setValue(r.branch); reportSheet.getRange(rowNum, 3).setValue(r.asset); reportSheet.getRange(rowNum, 5).setValue(timestamp); rowNum++; }); /* Merge and fill D.N. cell */ var dnRange = reportSheet.getRange(bStart, 4, bRows.length, 1); if (bRows.length > 1) dnRange.merge(); if (sharedDN) { dnRange.setFormula('=HYPERLINK("' + sharedDN + '","View D.N.")'); dnRange.setFontColor('#1155CC'); } else { dnRange.setValue('— No link yet —').setFontColor('#e67e22'); } dnRange.setVerticalAlignment('middle').setHorizontalAlignment('center'); /* Color rows */ reportSheet.getRange(bStart, 1, bRows.length, hdrs.length) .setBackground(secColor); }); } else { /* Individual asset rows */ rows.forEach(function(r) { reportSheet.getRange(rowNum, 1).setValue(r.section); reportSheet.getRange(rowNum, 2).setValue(r.branch); reportSheet.getRange(rowNum, 3).setValue(r.asset); if (r.dn && r.dn.startsWith('http')) { reportSheet.getRange(rowNum, 4) .setFormula('=HYPERLINK("' + r.dn + '","View D.N.")') .setFontColor('#1155CC'); } else { reportSheet.getRange(rowNum, 4).setValue('— No link yet —').setFontColor('#e67e22'); } reportSheet.getRange(rowNum, 5).setValue(timestamp); reportSheet.getRange(rowNum, 1, 1, hdrs.length).setBackground(secColor); rowNum++; }); } }); }); /* Any sections not in order */ Object.keys(sections).forEach(function(sec) { if (SECTION_ORDER.indexOf(sec) === -1) { var secColor = '#f8f9fa'; reportSheet.getRange(rowNum, 1, 1, hdrs.length) .merge().setValue(sec) .setFontWeight('bold').setBackground('#7f8c8d').setFontColor('#ffffff') .setHorizontalAlignment('center'); rowNum++; var groups = sections[sec]; Object.keys(groups).forEach(function(grp) { groups[grp].forEach(function(r) { reportSheet.getRange(rowNum, 1).setValue(r.section); reportSheet.getRange(rowNum, 2).setValue(r.branch); reportSheet.getRange(rowNum, 3).setValue(r.asset); if (r.dn && r.dn.startsWith('http')) { reportSheet.getRange(rowNum, 4).setFormula('=HYPERLINK("' + r.dn + '","View D.N.")').setFontColor('#1155CC'); } else { reportSheet.getRange(rowNum, 4).setValue('— No link yet —').setFontColor('#e67e22'); } reportSheet.getRange(rowNum, 5).setValue(timestamp); reportSheet.getRange(rowNum, 1, 1, hdrs.length).setBackground(secColor); rowNum++; }); }); } }); /* ── Final styling ── */ reportSheet.setFrozenRows(1); reportSheet.autoResizeColumns(1, hdrs.length); reportSheet.setColumnWidth(4, 120); var withLink = delivered.filter(function(r){ return r.dn && r.dn.startsWith('http'); }).length; var withoutLink = delivered.length - withLink; SpreadsheetApp.getUi().alert( 'Report generated!\n\n' + 'Total Delivered: ' + delivered.length + '\n' + 'With D.N. link: ' + withLink + '\n' + 'Missing D.N. link (orange): ' + withoutLink + '\n\n' + 'Check the "Delivered D.N. Report" tab.' ); }
